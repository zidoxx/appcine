  agregarProducto(){
    
    let data ={
       codigo:this.codigo,
       nombre:this.nombre,
       duracion:this.duracion,
       descripcion:this.descripcion,
       elenco:this.elenco,
	creadores: this.creadores,
	imagen: this.imagen,
	co_categoria: this.co_categoria,
	co_clasificacion: this.co_clasificacion
    }

    this.apiService.postProductos(data).subscribe(()=>{
	codigo="",
       	nombre="",
       	duracion="",

       	elenco="",
	creadores="",
	imagen="",
	co_categoria="",
	co_clasificacion="",
